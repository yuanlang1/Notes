# 多线程并发服务器

在使用线程模型开发服务器时需考虑以下问题：

1. 调整进程内最大文件描述符上限
2. 线程如有共享数据，考虑线程同步
3. 服务于客户端线程退出时，退出处理。（退出值，分离态）
4. 系统负载，随着链接客户端增加，导致其它线程不能及时得到`CPU`

## 1. `server`服务端

```c
#include <stdio.h>
#include <unistd.h>
#include <sys/stat.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <arpa/inet.h>
#include <pthread.h>
#include <string.h>
#include <ctype.h>
#include "wrap.h"

#define MAX_LEN 80
#define SERVER_PORT 8000

// 传递客户端信息
struct s_info{
    struct sockaddr_in clientaddr;
    int confd;
};

void *do_work(void *arg)
{
    int n, i;
    struct s_info *ts = (struct s_info *)arg;
    char buf[MAX_LEN];
    char str[INET_ADDRSTRLEN];  // 宏 16

    // 设置为分离态
    pthread_detach(pthread_self());

    while(1){
        // 读取数据
        n = Read(ts->confd, buf, MAX_LEN);
        if(n == 0){
            printf("the other side has been closed\n");
            break;  
         }
        // 输出接收到的客户端IP和端口
        printf("received from %s at port %d\n", 
            inet_ntop(AF_INET, &(*ts).clientaddr.sin_addr.s_addr, str, sizeof(str)),
            ntohs((*ts).clientaddr.sin_port));
        for(int i = 0; i < n; i++){
            buf[i] = toupper(buf[i]);
        }
        Write(ts->confd, buf, n);
    }
    Close(ts->confd);
}

int main(void)
{
    `int n, i;
    pthread_t tid;
    struct sockaddr_in serveraddr, clientaddr;
    int sockfd, confd;
    socklen_t clientlen;
    struct s_info ts[300];
    // 1. socket
    sockfd = Socket(AF_INET, SOCK_STREAM, 0);
    // 清空serveraddr
    bzero(&serveraddr, sizeof(serveraddr));
    // 地址协议族 IPv4
    serveraddr.sin_family = AF_INET;
    // IP地址
    serveraddr.sin_addr.s_addr = htonl(INADDR_ANY);
    // port 端口
    serveraddr.sin_port = htons(SERVER_PORT);
    // 2. bind
    Bind(sockfd, (struct sockaddr *)&serveraddr, sizeof(serveraddr));
    // 3. listen
    Listen(sockfd, 128);

    printf("Accepting connections ...\n");

    i = 0;
    while(1){
        clientlen = sizeof(clientaddr);
        confd = Accept(sockfd, (struct sockaddr *)&clientaddr, &clientlen);
        ts[i].clientaddr = clientaddr;
        ts[i].confd = confd;
        // 创建线程
        pthread_create(&tid, NULL, do_work, (void *)&ts[i]);
        i++;
    }

    Close(sockfd);

    return 0;
}
```

## 2. `client`客户端

```c
#include <stdio.h>
#include <unistd.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <string.h>
#include <pthread.h>
#include <arpa/inet.h>
#include "wrap.h"

#define MAX_LEN 80
#define SERVER_PORT 8000

int main(void)
{
    struct sockaddr_in serveraddr;
    char buf[MAX_LEN];
    char ipstr[] = "10.193.128.250";
    int confd, n;

    // 1. socket
    confd = Socket(AF_INET, SOCK_STREAM, 0);
    // 清空serveraddr
    bzero(&serveraddr, sizeof(serveraddr));
    // 地址族协议 IPv4
    serveraddr.sin_family = AF_INET;
    // IP地址
    inet_pton(AF_INET, ipstr, &serveraddr.sin_addr.s_addr);
    // port端口
    serveraddr.sin_port = htons(SERVER_PORT);
    // 连接服务器 
    Connect(confd, (struct sockaddr *)&serveraddr, sizeof(serveraddr));

    while(fgets(buf, MAX_LEN, stdin) != NULL){
        Write(confd, buf, strlen(buf));
        n = Read(confd, buf, MAX_LEN);
        if(n == 0){
            printf("the other side has been closed\n");
        }
        else{
            Write(STDOUT_FILENO, buf, n);
        }
    }
    Close(confd);

    return 0;
}
```
