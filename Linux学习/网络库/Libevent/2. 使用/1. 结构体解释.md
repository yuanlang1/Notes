# 结构体解释

## 1. `event`

```c
struct event {
    回调函数
    struct event_callback ev_evcallback;  

    定时器
    /* for managing timeouts */
    union {
        TAILQ_ENTRY(event) ev_next_with_common_timeout;
        int min_heap_idx;
    } ev_timeout_pos;

    定时事件的fd
    evutil_socket_t ev_fd;

    事件对象的基础结构
    struct event_base *ev_base;

    
    union {
        I/O事件队列
        /* used for io events */
        struct {
            LIST_ENTRY (event) ev_io_next;
            struct timeval ev_timeout;
        } ev_io;

        信号事件队列
        /* used by signal events */
        struct {
            LIST_ENTRY (event) ev_signal_next;
            short ev_ncalls;
            /* Allows deletes in callback */
                        short *ev_pncalls;
        } ev_signal;
    } ev_;

    事件类型
    short ev_events;
    
    记录当前激活事件的类型
    short ev_res;       /* result passed to event callback */
    
    指定定时器的超时值
    struct timeval ev_timeout;
};
```

## 2. `event_base`

```c
struct event_base {
    // 成员变量
    struct event_list** activequeues;   // 激活事件队列数组的指针
    int nactivequeues;                  // 激活事件的队列数量
    int event_count_active;             // 当前激活事件队列中的事件数量
    int event_count_added;              // 已添加到事件基础结构但尚未注册的事件数量
    int event_gotterm;                  // 表示是否收到终止事件，用于控制事件循环的终止
    int event_break;                    // 控制事件循环的暂停与恢复
    event* eventqueue;                  // 事件队列
    int event_count;                    // 当前事件基础结构中的事件数量
    struct timeval event_tv;            // 当前事件循环相关的时间信息

    // 更多成员和私有数据，省略...
};
```
